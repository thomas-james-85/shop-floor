-- DROP FUNCTION public.update_modified_timestamp();

CREATE OR REPLACE FUNCTION public.update_modified_timestamp()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
  NEW.updated_at = CURRENT_TIMESTAMP;
  RETURN NEW;
END;
$function$
